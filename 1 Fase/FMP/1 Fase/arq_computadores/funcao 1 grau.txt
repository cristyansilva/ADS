#include <Wire.h>
#include <LiquidCrystal_I2C.h>
#include <Keypad.h>

const byte linhas = 4;
const byte colunas = 4;

char tecla[linhas][colunas] = {
  {'1','2','3','A'},
  {'4','5','6','B'},
  {'7','8','9','C'},
  {'*','0','#','D'}
};

byte pinlinhas[linhas] = {16, 5, 4, 0};
byte pincolunas[colunas] = {2, 14, 12, 13};

Keypad myKeypad(makeKeymap(tecla), pinlinhas, pincolunas, linhas, colunas);
LiquidCrystal_I2C lcd(0x27, 16, 2);

String entrada = "";
float a = 0, b = 0, x = 0;
int etapa = 0; // 0 = inserir a, 1 = inserir b, 2 = inserir x

void setup() {
  Wire.begin(4, 5);
  lcd.init();
  lcd.backlight();

  lcd.setCursor(0, 0);
  lcd.print("Funcao 1o Grau:");
  delay(2000);
  lcd.clear();
  lcd.print("a = ");
}

void loop() {
  char key = myKeypad.getKey();

  if (key) {
    if (key >= '0' && key <= '9') {
      entrada += key;
      lcd.print(key);
    }

    // Tecla A: apaga entrada
    if (key == 'A') {
      entrada = "";
      lcd.clear();
      switch(etapa) {
        case 0: lcd.print("a = "); break;
        case 1: lcd.print("b = "); break;
        case 2: lcd.print("x = "); break;
      }
    }

    // Tecla #: confirma entrada e avanÃ§a
    if (key == '#') {
      float valor = entrada.toFloat();
      entrada = "";

      if (etapa == 0) {
        a = valor;
        etapa = 1;
        lcd.clear();
        lcd.print("b = ");
      }
      else if (etapa == 1) {
        b = valor;
        etapa = 2;
        lcd.clear();
        lcd.print("x = ");
      }
      else if (etapa == 2) {
        x = valor;
        float resultado = a * x + b;

        lcd.clear();
        lcd.setCursor(0, 0);
        lcd.print("f(x) = ");
        lcd.print(a); lcd.print("*");
        lcd.print(x); lcd.print("+");
        lcd.print(b);

        lcd.setCursor(0, 1);
        lcd.print("= ");
        lcd.print(resultado);

        etapa = 0; // reinicia
        delay(5000);
        lcd.clear();
        lcd.print("a = ");
      }
    }
  }
}
